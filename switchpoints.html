import pymc3 as pm
import numpy as np
import matplotlib.pyplot as plt


count_data = np.loadtxt("txtdata.csv")
n_count_data = len(count_data)

with pm.Model() as model:
    alpha = 1.0 / count_data.mean() #1/
    lambda_1 = pm.Exponential("lambda_1", alpha)
    lambda_2 = pm.Exponential("lambda_2", alpha)
    lambda_3= pm.Exponential("lambda_3", alpha)
    tau1 = pm.DiscreteUniform("tau1", lower=0, upper=n_count_data-1) 
    tau2=pm.DiscreteUniform("tau2", lower=tau1, upper=n_count_data)
    
    idx = np.arange(n_count_data)  # Index
    lambda_temp = pm.math.switch(tau1 >= idx , lambda_1,lambda_2)
    lambda_ = pm.math.switch(tau2 >= idx,lambda_temp,lambda_3 )


    observation = pm.Poisson("obs", lambda_, observed=count_data)

    trace = pm.sample(10000, tune=2000,cores=1,chains=2)

pm.traceplot(trace)
